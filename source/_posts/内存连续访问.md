比较内存连续访问和随机访问。

<!-- more -->

```cpp
#include<bits/stdc++.h>
using namespace std;
int a[10004][10004];
int main(){
	int n=10000;
	for(int t=1;t<=10;++t)for(int i=1;i<=n;++i)for(int j=1;j<=n;++j)a[i][j]=i|j^t;
}
```

0.7507s

```cpp
#include<bits/stdc++.h>
using namespace std;
int a[10004][10004];
int main(){
	int n=10000;
	for(int t=1;t<=10;++t)for(int i=1;i<=n;++i)for(int j=1;j<=n;++j)a[j][i]=i|j^t;
}
```

12.02s
